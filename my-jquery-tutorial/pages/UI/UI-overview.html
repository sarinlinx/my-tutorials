Add comparison check of 2 fields

Create vars in the function to obtain the value of each field.

Create if else statment to check if they match.

CSS is also removed and added using 'removeClass' and 'addClass'.

  • Example 2
.highlight {
  background-color: red;
}
                  
<div id="output">Output1 Content</div>
<!-- input fields -->
<input id="pswa" class="matchField" type="text" />
<br />
<input id="pswb" class="matchField" type="text" />
<div id="status"></div>
                               
 $(document).ready(function() {
  // only checking pswb because it's assumed the first field pswa is already filled in
  // on keyup is triggered, validate function is run
  $("#pswb").keyup(validate);

  function validate() {
    // obtain each value of the fields
    var psw1 = $("#pswa").val();
    var psw2 = $("#pswb").val();

    // run a comparison of the 2 fields
    if (psw1 === psw2) {
      $("#status").text("MATCHED");
      // remove the 'highlight' class
      $(".matchField").removeClass("highlight");
    } else {
      $("#status").text("NOT MATCH");
      // add the highlight class
      $(".matchField").addClass("highlight");
    }
  }
})

 $(document).ready(function() {

  // watch for click for checkbox 1
  $("#myCheck1").on("change", function() {
      console.log($(this));
      // if it's checked
      if ($(this).prop("checked")) {
        // creates popup
          var sure = confirm("Are you sure?");
          this.checked = sure;
      }
      // displays 'checked' div in HTML
      $("#output").html("Checked");
  });

  // add another event listener for 2nd checkbox
  $("#myCheck2").change(function() {
    // if it's checked
      if (this.checked) {
        // use 'show' to display
          $("#output").show();
      } else {
          $("#output").hide();
      }
  });

})

Create an autocomplete dropdown

Create an HTML search field.

Add an Array in jQuery.

User the 'autocomplete' function to check the Array.

When you type, a dropdown appears below the search field.

  • Example 8
<div class="ui-widget">
  <label for="searcher">Search Me:</label>
  <input id="searcher">
</div>

$(document).ready(function() {
  var foo = ["abc", "dcd", "eee"];
  // user 'autocomplete' on the source variable named 'foo'            
  $("#searcher").autocomplete({
      source: foo,
      // options
      multiple: true,
      // case check
      mustMatch: false
  }); 
});
