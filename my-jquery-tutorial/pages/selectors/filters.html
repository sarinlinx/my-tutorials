:has

Target all divs that contain a p

<div>
<p class="testClass">I am the only p surrounded by a div</p>
</div>
<p class="testingClass">p with class named 'testingClass'</p>
<p class="classTest">p with class named 'classTest'</p>
<button id="btn1">Click me</button>

$("#btn1").click(function() {
  $("div:has(p)").css({
    'border': '1px solid red'
  });
});
