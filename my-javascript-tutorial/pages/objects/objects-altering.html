Change a property in an Object

This changes a single property (by reference)

Sending an Object to a function's parameter creates a reference to the Object.

Any change to the reference makes a change to the actual Object.

This outputs twice:

  • { name: 'Andrew Mead', expenses: 2.5, income: 0 }
  • { name: 'Andrew Mead', expenses: 2.5, income: 0 }

This shows the new value within the function also permanenty changed the actual Object.

let myAccount = {
  name: 'Andrew Mead',
  expenses: 0,
  income: 0
}

//The account param is passing a reference to the myAccount Object in memory
let addExpense = function (account, amount) {
  //change expenses property
  account.expenses = account.expenses + amount
  //Print value of Object within this function
  console.log(account)
}

//Send the myAccount Object into the function as the first param
//Send 2.50 to the 2nd param
addExpense(myAccount, 2.50)
//Check the Object now and see the expenses property has changed
console.log(myAccount)

This does not change the Object since it breaks the reference

You can only change properties of the Object within the function.

If you assign the Object to a new Object, it DOES NOT change the original Object because the reference is then broken.

This outputs

  • {}
  • { name: 'Andrew Mead', expenses: 2.5, income: 0 }

Using the assignment operator on the param name creates a new local Object. It breaks the reference to the actual Object.

let myAccount = {
  name: 'Andrew Mead',
  expenses: 0,
  income: 0
}

let addExpense = function (account, amount) {
  //set account (the reference) to an empty Object
  account = {}
  //Print value of Object within this function
  console.log(account)
}

addExpense(myAccount, 2.50)
console.log(myAccount)

Changing the property by reference (without a function)

By assigning a variable to the Object, you create a reference to the Object.

This outputs

  • { name: 'Andrew Mead', expenses: 2.5, income: 1000 }
  • { name: 'Andrew Mead', expenses: 2.5, income: 1000 }
let myAccount = {
  name: 'Andrew Mead',
  expenses: 0,
  income: 0
}

//Create a reference (binding) to the original Object
let otherAccount = myAccount
//Change a property of this reference
otherAccount.income = 1000

let addExpense = function (account, amount) {
  account.expenses = account.expenses + amount
  console.log(account)
}

addExpense(myAccount, 2.50)
console.log(myAccount)
