Changing attributes

Syntax is: var result = elementName.hasAttribute(name);

'result' holds either TRUE or FALSE.

'name' is a string of the name of the attribute.

Check if an element has a specific attribute (hasAttribute)

Checking by the ID

This checks if an element with an id of 'headerID' also contains a 'class' attribute.

This returns TRUE.

View the following link for further details.

  • hasAttribute
<header id="headerID" class="masthead">header here</header>
<button onclick="myFunction()">Click to change</button>

function myFunction() {
  //first get the element
  var hasAttr = document.getElementById("headerID").hasAttribute("class");
  alert(hasAttr);
}

Checking using querySelector with alert()

This checks if the 'header' element contains a 'class' attribute.

This returns TRUE.

View the following link for further details.

<header class="masthead">header here</header>
<button onclick="myFunction()">Click to change</button>

function myFunction() {
  //first get the element
  var hasAttr = document.querySelector("header").hasAttribute("class");
  alert(hasAttr);
}

Checking using querySelector with console

This checks if the inline 'style' element exists.

This returns 'true' in the console.

<div class="cta" style="color: red;">
  <p>Text</p>
</div>

document.querySelector(".cta").hasAttribute("style");
true

getAttribute()

Return the attribute names of an element as an Array of strings

This first looks for the element with an id of 'paragraphID'.

It then checks if the attribute 'class' also exists for this same element.

If so, it returns the string name of that attribute.

This example returns "myClass".

View the following for more info:

  • getAttribute
<p id="paragraphID" class="myClass"><a href="#">header here</a></p>
<button onclick="myFunction()">Click to change</button>

function myFunction() {
  var hasAttr = document.getElementById("paragraphID").getAttribute("class");
  alert(hasAttr);
}

Another example.

This returns "_blank".

<a id="myAnchor" href="#" target="_blank">Attr object</a>
<button onclick="myFunction()">Click to change</button>

function myFunction() {
  var hasAttr = document.getElementById("myAnchor").getAttribute("target");
  alert(hasAttr);
}

Retrieving using querySelector with console

This returns the value of the inline 'style' if it exists.

This returns "color: red;" in the console.

<div class="cta" style="color: red;">
<p>Text</p>
</div>

document.querySelector(".cta").getAttribute("style");
"color: red;"

setAttribute()

Sets the value of an attribute on the specified element

Sets the value of an attribute on the specified element. If the attribute already exists, the value is updated; otherwise a new attribute is added with the specified name and value.

The retrieves the element with a class of 'paragraphID', then adds a new class of 'mynewClass'.

  • setAttribute
<p id="paragraphID" class="myClass"><a href="#">header here</a></p>
<button onclick="myFunction()">Click to change</button>

function myFunction() {
  document.getElementById("paragraphID").setAttribute("class", "mynewClass");
  var getNewClass = document.querySelector("#paragraphID").classList;
  alert(getNewClass);
}

This retrieves the element with an id of 'myAnchor'.

It then adds a new attribute of 'href' with a value of 'example.com'.

<a id="myAnchor">A Link: Go to w3schools.com</a>
<button onclick="myFunction()">Click to change</button>

function myFunction() {
  document.getElementById("myAnchor").setAttribute("href", "https://www.example.com");
}

Changing and Inline CSS style

This only works to change CSS if it's written Inline.

This chanages the RED inline color to GREEN.

<div class="cta" style="color: red;">
  <p>Text</p>
</div>

document.querySelector(".cta").setAttribute("style","color: green");

removeAttribute()

removeAttribute() removes the attribute with the specified name from the element.

  • removeAttribute
<p id="paragraphID" class="myClass"><a href="#">header here</a></p>
<button onclick="myFunction()">Click to change</button>

//Check original class list for element
var classList = document.querySelector("#pID").classList;
alert(classList);

function myFunction() {
    document.getElementsByTagName("p")[0].removeAttribute("class");
    //Check if the original classes exist
    var newclassList = document.querySelector("#pID").classList;
    alert(newclassList);
}

Remove an inline style using querySelector with console

This remove the inline 'style' attribute if it exists.

You can then test if it's been removed by using .hasAttribute().

<div class="cta" style="color: red;">
<p>Text</p>
</div>

document.querySelector(".cta").removeAttribute("style");
document.querySelector(".cta").hasAttribute("style");
