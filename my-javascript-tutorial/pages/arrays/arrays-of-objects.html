Create an array of 'Course' Objects

This creates an Array of Objects named 'Course'.

You must first create the Object constructor. Only then can objects be created in the Array.

This outputs:
0: Object { title: "JavaScript Essential Training", instructor: "Teacher Name", level: 1, … }

1: Object { title: "New course name", instructor: "other teacher Name", level: 3, … }
​

//Create the Object constructor first
function Course(title,instructor,level,views) {
    this.title = title;
    this.instructor = instructor;
    this.level = level;
    this.views = views;
    this.updateViews = function() {
        return ++this.views;
    };
}

//You can now add/construct new Objects within the Array
var courses = [
  new Course("JavaScript Essential Training", "Teacher Name", 1, 0),
  new Course("New course name", "other teacher Name", 3, 1234)
]

console.log(courses);
console.log(courses[1].instructor);
//You can run the function in the Object constructor to update the Course view #
console.log(courses[1].updateViews());
