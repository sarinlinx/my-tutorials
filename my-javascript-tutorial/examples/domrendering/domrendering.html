//Create Array with data
  let notes = [{
      title: 'my next trip',
      body: 'I would like to go to Spain',
      completed: true
  }, {
      title: 'Habbits to work on',
      body: 'Exercise. Eating a bit better.',
      completed: true
  }, {
      title: 'Office modification',
      body: 'Get a new seat',
      completed: false
  }]

  //Create var to store function that loops through array
  //and runs the append function generateTodoDOM
  const savedTodos = function(x) {
      x.forEach(function(i) {
          document.querySelector('#content').appendChild(generateTodoDOM(i))
      })
  }

  //Print content to page
  const generateTodoDOM = function(i) {

  //Create a root div tag, this is what everything is appended to
  const todoEl = document.createElement('div')
  //Create checkbox, span, and button
  const checkbox = document.createElement('input')
  const todoText = document.createElement('span')
  const removeButton = document.createElement('button')

  //set the checkbox attribute, span and button content
  checkbox.setAttribute('type', 'checkbox')
  todoText.textContent = i.title
  removeButton.textContent = 'x'

  //Print to page
  todoEl.appendChild(checkbox)
  todoEl.appendChild(todoText)
  todoEl.appendChild(removeButton)

  return todoEl
}

//Run savedTodos()
savedTodos(notes)
