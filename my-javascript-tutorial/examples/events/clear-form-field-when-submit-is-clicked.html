Clear form field when submit button is clicked

HTML

Type into the form field and click the submit button.

Your text appears in the console.

The text you entered into the field is reset to the placeholder.

<form id="name-form">
  <input type="text" placeholder="First name" name="firstName">
  <button>Submit</button>
</form>

// target element with ID of #name-form and listen for 'submit' event
document.querySelector('#name-form').addEventListener('submit', function(e) {
  // cancel the default behavior of the form
  // prevents full page refresh or adding variables to the URL string
  e.preventDefault()
    // e.target gets you the DOM element of the form
    // form DOM elements have a property called .elements
    // .elements allows you to access all fields
    // target the value fields and get its value
    console.log(e.target.elements.firstName.value)
    // clear the form field after submit is clicked
    e.target.elements.firstName.value = ''
})
